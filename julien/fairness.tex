\section{Les donn\'ees}

\begin{frame}{Le German Credit Dataset}
\begin{columns}[T]
\begin{column}{0.55\textwidth}
\textbf{UCI Statlog} (Hofmann, 1994)
\begin{itemize}
    \item 1\,000 demandes de cr\'edit
    \item 20 attributs (7 num., 13 cat.)
    \item Cible : d\'efaut de paiement
\end{itemize}

\vspace{0.5em}
\textbf{Pourquoi ce dataset ?}
\begin{itemize}
    \item Scoring cr\'edit = d\'ecision \`a fort impact humain
    \item Risque de discrimination historique
    \item Cas d'\'ecole en IA responsable
\end{itemize}
\end{column}
\begin{column}{0.42\textwidth}
\centering
\small
\begin{tabular}{lc}
\toprule
\textbf{Stat} & \textbf{Valeur} \\
\midrule
Bon payeur (0) & 700 (70\%) \\
Mauvais (1) & 300 (30\%) \\
\midrule
Hommes & 690 (69\%) \\
Femmes & 310 (31\%) \\
\midrule
Adultes ($>$25) & 810 (81\%) \\
Jeunes ($\leq$25) & 190 (19\%) \\
\bottomrule
\end{tabular}

\vspace{0.5em}
\scriptsize D\'es\'equilibre de la cible ET des groupes
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Choix : deux attributs sensibles}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{\textcolor{accent}{Genre}}
\begin{itemize}
    \item Extrait de \texttt{personal\_status\_sex}
    \item Privil\'egi\'e : homme
    \item Biais mod\'er\'e dans les donn\'ees
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{\textcolor{fair}{\^Age}}
\begin{itemize}
    \item Binaris\'e au seuil de 25 ans
    \item Privil\'egi\'e : adulte
    \item Biais plus marqu\'e (12 points)
\end{itemize}
\end{column}
\end{columns}

\vspace{0.8em}
\begin{block}{D\'ecision de conception}
Les deux colonnes (\texttt{personal\_status\_sex}, \texttt{age\_in\_years})
sont \textbf{retir\'ees des features} $\Rightarrow$ pas de discrimination \emph{directe}.

\vspace{0.3em}
Mais des \textbf{proxies} (variables corr\'el\'ees) peuvent encoder indirectement
l'information sensible. On le v\'erifiera via SHAP.
\end{block}
\end{frame}

\begin{frame}{Mod\`ele de base}
\textbf{R\'egression logistique} :
$P(\text{d\'efaut}=1 \mid x) = \sigma(w^\top x + b)$

\vspace{0.5em}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Entra\^inement}
\begin{itemize}
    \item \texttt{sklearn.LogisticRegression}, solver \texttt{liblinear}, $C{=}1$
    \item L2 r\'egularis\'ee, \texttt{max\_iter}\,=\,1000
    \item Convergence v\'erifi\'ee : log-loss val 0.56 d\`es 10 it\'erations (seuil $\sim$0.5 attendu)
    \item Seuil : max balanced accuracy (val) $\to$ $\tau{=}0.18$
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Pr\'etraitement (\texttt{ColumnTransformer})}
\begin{itemize}
    \item \texttt{StandardScaler} (num.) + \texttt{OneHotEncoder} (cat.)
    \item 56 features apr\`es encodage
    \item Genre et \^age retir\'es (choix initial)
\end{itemize}
\end{column}
\end{columns}

\vspace{0.5em}
\begin{block}{Choix des m\'etriques}
Dataset d\'es\'equilibr\'e (70/30) $\Rightarrow$ accuracy simple trompeuse. \\
On utilise \textbf{AUC ROC} (invariant au seuil), \textbf{balanced accuracy}
$= \frac{1}{2}(\text{TPR} + \text{TNR})$, et \textbf{ECE} (calibration).
\end{block}
\end{frame}

\section{\'Equit\'e}

\begin{frame}{D\'efinitions (cours fairness)}
Soit $S$ l'attribut sensible, $Y$ la cible, $\hat{Y}$ la pr\'ediction.

\vspace{0.5em}
\begin{tabular}{lll}
\toprule
\textbf{Crit\`ere} & \textbf{Signification} & \textbf{On mesure} \\
\midrule
Demographic Parity & M\^eme taux de s\'election & $|P(\hat{Y}{=}1|S{=}0) - P(\hat{Y}{=}1|S{=}1)|$ \\
Equal Opportunity & M\^eme TPR & $|\text{TPR}_{S=0} - \text{TPR}_{S=1}|$ \\
Equalized Odds & M\^eme TPR + FPR & EOpp + m\^eme FPR \\
\bottomrule
\end{tabular}

\vspace{0.8em}
\begin{alertblock}{Th\'eor\`eme d'impossibilit\'e (Chouldechova 2017, Kleinberg 2016)}
On \textbf{ne peut pas} satisfaire DP, EO et calibration simultan\'ement
(sauf si les taux de base sont \'egaux entre groupes).
$\Rightarrow$ Il faut \textbf{choisir} ses compromis.
\end{alertblock}
\end{frame}

\begin{frame}{Reweighing (pr\'e-traitement)}
\textbf{Id\'ee} : repondérer les exemples pour rendre $S \perp Y$ dans le train.

$$w_i = \frac{P(S{=}s_i) \cdot P(Y{=}y_i)}{P(S{=}s_i, Y{=}y_i)}$$

\begin{columns}[T]
\begin{column}{0.55\textwidth}
Si les femmes sans d\'efaut sont sous-repr\'esent\'ees
$\Rightarrow$ on augmente leur poids.

\vspace{0.5em}
Un mod\`ele par attribut sensible.
\end{column}
\begin{column}{0.42\textwidth}
\centering\small
\begin{tabular}{lcc}
\toprule
Groupe & $Y{=}0$ & $Y{=}1$ \\
\midrule
Privil. & $w \approx 1$ & $w > 1$ \\
Non-priv. & $w > 1$ & $w \approx 1$ \\
\bottomrule
\end{tabular}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Seuils par groupe (post-traitement)}
\textbf{Id\'ee} : ajuster le seuil de classification par groupe sur la validation.

\vspace{0.5em}
\begin{itemize}
    \item Seuil global $\tau$ $\to$ seuils $\tau_{g_0}, \tau_{g_1}$
    \item Crit\`ere : \'egaliser le taux de s\'election (DP)
    \item Calibr\'e sur la validation ($\sim$200 exemples)
\end{itemize}

\vspace{0.5em}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{exampleblock}{Avantage}
Pas de r\'e-entra\^inement.
\end{exampleblock}
\end{column}
\begin{column}{0.48\textwidth}
\begin{alertblock}{Limite}
Peu de donn\'ees de calibration $\Rightarrow$ bruit.
\end{alertblock}
\end{column}
\end{columns}
\end{frame}

\end{itemize}
\end{frame}

\section{Interpr\'etabilit\'e}

\begin{frame}{SHAP lin\'eaire exact}
\textbf{Question} : quelles features d\'eterminent les d\'ecisions ?

\vspace{0.3em}
Pour un mod\`ele lin\'eaire, les \textbf{valeurs de Shapley} sont exactes :
$$\phi_i(x) = w_i \cdot \left(x_i - \mathbb{E}_{\text{train}}[x_i]\right)$$

\vspace{0.3em}
\textbf{Top 5 features} (SHAP baseline, via \texttt{shap.LinearExplainer}) :
\begin{enumerate}
    \item \texttt{checking\_status} --- compte courant
    \item \texttt{credit\_history} --- historique
    \item \texttt{installment\_rate} --- taux d'\'ech\'eance
    \item \texttt{savings\_account\_bonds} --- \'epargne
    \item \texttt{purpose}, \texttt{credit\_amount}, \texttt{duration\_in\_month}
\end{enumerate}

\vspace{0.3em}
Compl\'et\'e par l'\textbf{importance par permutation} (\texttt{sklearn.inspection.permutation\_importance}, chute d'AUC quand on m\'elange une colonne). Les deux m\'ethodes convergent sur le top.
\end{frame}

\begin{frame}{D\'etection de proxies (approche Charlotte)}
\textbf{Question} : peut-on pr\'edire l'attribut sensible \`a partir des autres features ?

\vspace{0.3em}
\textbf{M\'ethode} : r\'egression logistique \texttt{features $\to$ attribut sensible}, AUC mesure le signal proxy multivari\'e.

\vspace{0.3em}
\centering
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lcl}
\toprule
\textbf{Attribut} & \textbf{AUC proxy} & \textbf{Interpr\'etation} \\
\midrule
Genre & 0.69 & signal proxy mod\'er\'e \\
\^Age & \textcolor{alertc!80!black}{\textbf{0.86}} & \textcolor{alertc!80!black}{\textbf{tr\`es fortement reconstructible}} \\
\bottomrule
\end{tabular}

\vspace{0.5em}
\begin{alertblock}{Le\c{c}on}
Retirer \texttt{age\_in\_years} ne suffit pas \`a \og cacher \fg{} l'\^age : il est encod\'e dans \texttt{housing}, \texttt{employment}, \texttt{job}\ldots

\textbf{Recommandation Charlotte :} ne pas retirer les attributs sensibles, les utiliser explicitement comme pond\'erations (\textit{fairness through awareness}).
\end{alertblock}
\end{frame}

\begin{frame}{SHAP : avant vs apr\`es retrait des attributs sensibles}
\centering
\includegraphics[width=0.92\linewidth]{../outputs/08_shap_compare.png}

\section{Quantification d'incertitude}

\begin{frame}{Bootstrap (substitut \`a la robustesse adversariale)}
\textbf{Choix m\'ethodologique} (feedback Charlotte) : la robustesse adversariale est remplac\'ee par une \textbf{quantification d'incertitude par bootstrap} (cf. cours \textit{Robustness and Uncertainty}).

\vspace{0.3em}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Protocole}
\begin{itemize}
    \item $B = 200$ mod\`eles entra\^in\'es sur des \'echantillons tir\'es avec remplacement
    \item Quantiles 2.5\,\%/97.5\,\% des $B$ valeurs $\to$ IC95\,\%
    \item Sur AUC, |DP|, |EO|, et scores individuels
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{R\'esultats}
\begin{itemize}
    \item AUC : 0.764 \textbf{[0.72, 0.80]}
    \item $|\Delta\text{DP}|$ \^age : 0.10 \textbf{[0.02, 0.20]}
    \item $|\Delta\text{EO}|$ \^age : 0.09 \textbf{[0.02, 0.19]}
\end{itemize}
\end{column}
\end{columns}

\vspace{0.5em}
\begin{alertblock}{Lecture critique}
Sur 1000 lignes, les IC sur les biais d\'epassent 0.15 $\Rightarrow$ \textbf{prudence statistique}, les conclusions de discrimination doivent \^etre nuanc\'ees.
\end{alertblock}
\end{frame}

\begin{frame}{Calibration : ECE et Platt scaling}
\textbf{ECE} (cours \textit{Robustness and Uncertainty}) :
$\text{ECE} = \sum_{b} \frac{|B_b|}{n} |\text{acc}_b - \text{conf}_b|$

\vspace{0.5em}
\centering
\begin{tabular}{lc}
\toprule
\textbf{Mod\`ele} & \textbf{ECE} \\
\midrule
Baseline (R\'eg.\ logist.) & 0.082 \\
Platt scaling (\texttt{CalibratedClassifierCV}) & 0.080 \\
\bottomrule
\end{tabular}

\vspace{0.5em}
$\Rightarrow$ La r\'egression logistique est \textbf{naturellement bien calibr\'ee}, le gain marginal du Platt scaling est minime.
\end{frame}

\section{Synth\`ese}

\begin{frame}{Ce qu'on a appris}
\begin{enumerate}
    \item \textbf{Les donn\'ees sont biais\'ees sur l'\^age} (younger 42\,\% vs older 27\,\% de d\'efaut), pas sur le genre apr\`es retrait de \texttt{personal\_status\_sex}.

    \item \textbf{Retirer l'attribut sensible ne suffit pas} : la r\'egression logistique \texttt{features $\to$ \^age} a un \textbf{AUC = 0.86}. L'\^age est largement encod\'e dans les autres variables (\texttt{housing}, \texttt{employment}\ldots).

    \item \textbf{Compromis DP/EO in\'evitable} : sur l'\^age, le post-processing r\'eduit $|\Delta\text{DP}|$ (0.125 $\to$ 0.060) mais \textbf{double} $|\Delta\text{EO}|$ (0.066 $\to$ 0.135). Th\'eor\`eme d'impossibilit\'e.

    \item \textbf{L'interpr\'etabilit\'e \'eclaire l'\'equit\'e} : SHAP + permutation + r\'egression sur l'attribut sensible = d\'etection multi-niveaux des proxies.

    \item \textbf{Bootstrap = robustesse statistique} : IC95\,\% large sur les biais (0.02 \`a 0.20 pour $|\Delta\text{DP}|$ \^age) $\Rightarrow$ prudence m\'ethodologique.

    \item \textbf{Recommandation Charlotte :} pour aller plus loin, conserver l'attribut sensible et appliquer une mitigation explicite (\textit{fairness through awareness}, \texttt{ExponentiatedGradient}, transport optimal).
\end{enumerate}

\end{frame}
